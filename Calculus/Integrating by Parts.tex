\section{Introduction to Calculus: Integration by Parts}

\begin{frame}
\[ \int xe^xdx \]

\begin{itemize}
\item Let u = x, Therefore we can say 
\[ \frac{du}{dx}=1\]

and furthermore $du=dx$

\item Let $dv = e^xdx$

\[ v =\int dv = \int e^xdx =e^x \]

\end{itemize}
\end{frame}
