\textbf{Exercise}: Find the \textit{\textbf{discounted value}} of $\$2,500$ due in four months if the rate is 12\%.
\bigskip
\begin{itemize}
\item S = 2,500
\item r = 12\ %
\item t = ${4\over 12}$
\end{itemize}
\end{frame}

\[\mbox{ANSWER:    }\;\;  P = \$2403.85\]
